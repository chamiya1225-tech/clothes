function load(key, def) {
  try { const v = localStorage.getItem(key); return v ? JSON.parse(v) : def; } catch { return def; }
}
function save(key, val) {
  try { localStorage.setItem(key, JSON.stringify(val)); } catch {}
}

function daysSince(d) { if(!d) return 999; return Math.floor((Date.now()-new Date(d))/86400000); }
function catIcon(c) { return CATEGORIES[c]?.icon || '📦'; }
function totalValue() { return items.reduce((a,b)=>a+(Number(b.price)||0),0); }
function avgWear() { return items.length ? Math.round(items.reduce((a,b)=>a+b.wearCount,0)/items.length) : 0; }
function idleItems() { return items.filter(i=>daysSince(i.lastWorn)>60); }

function showPage(p) {
  currentPage = p;
  ['wardrobe','outfits','wishlist','stats'].forEach(pg => {
    document.getElementById('page-'+pg).classList.toggle('hidden', pg!==p);
  });
  document.querySelectorAll('.nav-btn').forEach((btn,i) => {
    btn.classList.toggle('active', ['wardrobe','outfits','wishlist','stats'][i]===p);
  });
  if(p==='wardrobe') renderWardrobe();
  if(p==='outfits') renderOutfits();
  if(p==='wishlist') renderWishlist();
  if(p==='stats') renderStats();
}

function renderWardrobe() {
  document.getElementById('wardrobe-sub').textContent = `${items.length} 件单品 · 共 ¥${totalValue().toLocaleString()}`;
  renderCatFilters();
  filterItems();
}

function renderCatFilters() {
  const el = document.getElementById('cat-filters');
  const cats = ['全部', ...Object.keys(CATEGORIES)];
  el.innerHTML = cats.map(c => {
    const icon = CATEGORIES[c]?.icon || '';
    return `<button class="chip ${selectedCat===c?'chip-active':'chip-inactive'}" onclick="selectCat('${c}')">${icon} ${c}</button>`;
  }).join('');
}

function selectCat(c) {
  selectedCat = c; selectedSub = '全部';
  renderCatFilters();
  const subEl = document.getElementById('sub-filters');
  if(c==='全部') { subEl.style.display='none'; subEl.innerHTML=''; }
  else {
    subEl.style.display='flex';
    const subs = ['全部', ...(CATEGORIES[c]?.sub||[])];
    subEl.innerHTML = subs.map(s =>
      `<button class="chip ${selectedSub===s?'chip-active':'chip-inactive'}" onclick="selectSub('${s}')">${s}</button>`
    ).join('');
  }
  filterItems();
}

function selectSub(s) {
  selectedSub = s;
  document.getElementById('sub-filters').querySelectorAll('.chip').forEach((btn,i) => {
    const subs = ['全部', ...(CATEGORIES[selectedCat]?.sub||[])];
    btn.className = `chip ${subs[i]===s?'chip-active':'chip-inactive'}`;
  });
  filterItems();
}

function filterItems() {
  const search = document.getElementById('search-input').value.toLowerCase();
  const filtered = items.filter(i => {
    const catOk = selectedCat==='全部' || i.category===selectedCat;
    const subOk = selectedSub==='全部' || i.sub===selectedSub;
    const srOk = !search || i.name.toLowerCase().includes(search) || (i.brand||'').toLowerCase().includes(search);
    return catOk && subOk && srOk;
  });
  const grid = document.getElementById('items-grid');
  const noEl = document.getElementById('no-items');
  if(!filtered.length) { grid.innerHTML=''; noEl.classList.remove('hidden'); return; }
  noEl.classList.add('hidden');
  grid.innerHTML = filtered.map(item => {
    const idle = daysSince(item.lastWorn)>60;
    const imgHtml = item.image
      ? `<img src="${item.image}" alt="${item.name}" style="width:100%;height:100%;object-fit:cover" />`
      : `<span style="font-size:40px">${catIcon(item.category)}</span>`;
    const bg = item.image ? 'transparent' : item.color+'22';
    return `<div class="card" onclick="openItemDetail(${item.id})">
      <div class="card-img" style="background:${bg};padding:0">${imgHtml}</div>
      ${idle?'<div class="idle-badge">闲置</div>':''}
      <div class="card-body">
        <div class="card-name">${item.name}</div>
        <div class="card-brand">${item.brand||''}</div>
        <div style="margin-top:6px;display:flex;align-items:center">
          <span class="color-dot" style="background:${item.color}"></span>
          <span style="font-size:10px;color:#9A9088">${item.colorName||''}</span>
        </div>
        <div style="font-size:10px;color:#9A9088;margin-top:4px">穿了 ${item.wearCount} 次</div>
      </div>
    </div>`;
  }).join('');
}

function openItemDetail(id) {
  const item = items.find(i=>i.id===id);
  if(!item) return;
  currentItemId = id;
  document.getElementById('detail-name').textContent = item.name;
  document.getElementById('detail-meta').textContent = `${item.brand||''} · ${item.category} · ${item.sub}`;
  const imgWrap = document.getElementById('detail-img-wrap');
  if(item.image) {
    imgWrap.innerHTML = `<img src="${item.image}" style="width:72px;height:72px;object-fit:cover;border-radius:8px;border:1px solid #E8E2D8;flex-shrink:0" />`;
  } else {
    imgWrap.innerHTML = `<div style="width:72px;height:72px;border-radius:8px;background:${item.color}33;display:flex;align-items:center;justify-content:center;font-size:36px;flex-shrink:0">${catIcon(item.category)}</div>`;
  }
  const infoData = [
    ['材质',item.material],['版型',item.fit],
    ['购入时间',item.buyDate],['购入价格',item.price?`¥${item.price}`:'—'],
    ['适合季节',(item.season||[]).join('、')],['状态',item.status],
    ['穿着次数',`${item.wearCount} 次`],
    ['次均成本',item.price&&item.wearCount>0?`¥${Math.round(item.price/item.wearCount)}`:'—'],
  ];
  document.getElementById('detail-info').innerHTML = infoData.map(([k,v])=>`
    <div class="info-cell"><div class="info-key">${k}</div><div class="info-val">${v||'—'}</div></div>`).join('');
  document.getElementById('detail-wash-method').textContent = item.washMethod||'';
  document.getElementById('detail-wash-note').textContent = item.washNote||'';
  const related = outfits.filter(o=>o.items.includes(id));

    ? `<div class="section-title">出现在以下搭配中</div>${related.map(o=>`<div style="font-size:12px;padding:6px 0;border-bottom:1px solid #F0EDE8;color:#5A5248">✦ ${o.name} <span style="color:#9A9088">· ${o.scene}</span></div>`).join('')}<div style="margin-bottom:16px"></div>`

function initAddItemModal() {
  const catSel = document.getElementById('ni-cat');

function updateSubCat() {

  const subSel = document.getElementById('ni-sub');

function toggleSeason(s, btn) {
  const idx = selectedSeasons.indexOf(s);
  if(idx>-1) { selectedSeasons.splice(idx,1); btn.className='chip chip-inactive'; }
  else { selectedSeasons.push(s); btn.className='chip chip-active'; }
}

function handleImageUpload(e) {
  const file = e.target.files[0];
  if(!file) return;
  const reader = new FileReader();
  reader.onload = ev => {
    imageDataUrl = ev.target.result;
    const zone = document.getElementById('upload-zone');
    zone.innerHTML = `<img src="${imageDataUrl}" alt="preview" />`;
    zone.classList.add('has-img');
    document.getElementById('clear-img-btn').classList.remove('hidden');
  };
  reader.readAsDataURL(file);
}

function clearImage() {
  imageDataUrl = null;
  const zone = document.getElementById('upload-zone');
  zone.innerHTML = `<div style="font-size:28px;margin-bottom:6px">📷</div><div style="font-size:12px;color:#9A9088">点击上传照片</div><div style="font-size:11px;color:#B8B2A8;margin-top:2px">支持 JPG、PNG、WEBP</div>`;
  zone.classList.remove('has-img');
  document.getElementById('clear-img-btn').classList.add('hidden');
  document.getElementById('img-upload').value = '';
}

function addItem() {
  const name = document.getElementById('ni-name').value.trim();
  if(!name) { alert('请填写单品名称'); return; }
  const item = {
    id: Date.now(),
    name,
    brand: document.getElementById('ni-brand').value.trim(),
    category: document.getElementById('ni-cat').value,
    sub: document.getElementById('ni-sub').value,
    color: document.getElementById('ni-color').value,
    colorName: document.getElementById('ni-color-name').value.trim(),
    fit: document.getElementById('ni-fit').value,
    material: document.getElementById('ni-material').value.trim(),
    buyDate: document.getElementById('ni-date').value,
    price: Number(document.getElementById('ni-price').value)||0,
    washMethod: document.getElementById('ni-wash').value,
    washNote: document.getElementById('ni-wash-note').value.trim(),
    season: [...selectedSeasons],
    status: '在穿',
    wearCount: 0,
    lastWorn: null,
    image: imageDataUrl,
  };
  items.push(item);
  save('wardrobe_items', items);
  closeAddItem();
  filterItems();
  document.getElementById('wardrobe-sub').textContent = `${items.length} 件单品 · 共 ¥${totalValue().toLocaleString()}`;
}

}

function openModal(id) {
  if(id==='add-item-modal') initAddItemModal();
  if(id==='add-outfit-modal') initOutfitModal();
  document.getElementById(id).classList.remove('hidden');
}

function initOutfitModal() {
  selectedOutfitItems = [];
  const sel = document.getElementById('outfit-items-selector');
  sel.innerHTML = items.map(it=>
    `<button class="chip chip-inactive" onclick="toggleOutfitItem(${it.id},this)">${catIcon(it.category)} ${it.name}</button>`
  ).join('');
}

function toggleOutfitItem(id, btn) {
  const idx = selectedOutfitItems.indexOf(id);
  if(idx>-1) { selectedOutfitItems.splice(idx,1); btn.className='chip chip-inactive'; }
  else { selectedOutfitItems.push(id); btn.className='chip chip-active'; }
}

function addOutfit() {
  const name = document.getElementById('no-name').value.trim();
  if(!name) { alert('请填写搭配名称'); return; }
  const tagsRaw = document.getElementById('no-tags').value.trim();
  const outfit = {
    id: Date.now(),
    name,
    scene: document.getElementById('no-scene').value,
    rating: Number(document.getElementById('no-rating').value),
    items: [...selectedOutfitItems],
    note: document.getElementById('no-note').value.trim(),
    date: new Date().toISOString().slice(0,10),
    tags: tagsRaw ? tagsRaw.split(/[，,]/).map(t=>t.trim()).filter(Boolean) : [],
  };
  outfits.push(outfit);
  save('wardrobe_outfits', outfits);
  closeModal('add-outfit-modal');
  document.getElementById('no-name').value='';
  document.getElementById('no-note').value='';
  document.getElementById('no-tags').value='';
  renderOutfits();
}
